1 /**
  2  * Dictionary-based persistent storage.
  3 
  4  * @description As this class relies on Titanium.App.Properties, for each value Foundation will store an extra value
  5  * to keep reference of the value's type. This allows you to get and set values without worrying of data types and
  6  * performances.
  7  */
  8 Foundation.PersistentStorage = /** @lends Foundation.PersistentStorage# */{
  9 	_watch: {},
 10 	Event: /** @lends Foundation.PersistentStorage.Event# */ {
 11 		/**
 12 		 * Event name to use when watching value changes.
 13 		 * @constant
 14 		 */
 15 		CHANGE: 'Foundation.PersistentStorage.change'
 16 	},
 17 	
 18 	/**
 19 	 * @name Foundation.PersistentStorage.change
 20 	 * Event triggered when a value changes. Fired only on demand, that is, only if one or more key listeners
 21 	 * are registered.
 22 	 * @event
 23 	 * @param {Object} e Event object
 24 	 * @param {string} e.type Type of the event (currently only set is supported)
 25 	 * @param {string} e.keySelector The selector query that matched the key. For example, if the key 'cat'
 26 	 *								 has changed and the regular expression /(b|c)at/ matched, this parameter will contain
 27 	 * 								 the string /(b|c)at/. You can use this string to re-create the matching regex.
 28 	 * @param {string} e.key		 The key for which the event has occurred. In the example above, this will be 'cat'.
 29 	 * @param {mixed}  e.value		 The new value.
 30 	 */
 31 
 32 	/**
 33 	 * Gets one or more values safely.
 34 	 * @param {mixed} key The key to remove. If string, the key will be removed, if present. You can
 35 	 *					   provide a regex that will delete all the matching keys.
 36 	 * @param {boolean} [includeKey=false] True to return an object with two properties: key (the key for this value)
 37 	 *                  and value (the value).
 38 	 * @return {mixed} If you provided a string key:
 39 	 * 				<ul>
 40 	 *					<li>If the key is present, the value will be returned.</li>
 41 	 *					<li>If the key is present and includeKey is true, an object will be returned (see the description
 42 	 *                  for includeKey).</li>
 43 	 *					<li>If the key is not present, null will be returned.</li>
 44 	 *				</ul>
 45 	 *				If you provided a regex:
 46 	 * 				<ul>
 47 	 *					<li>If the regex matches one or more keys, an array of values will be returned.</li>
 48 	 *					<li>If the regex matches one or more keys and includeKey is true, an array of objects
 49 	 *                  will be returned (see the description for includeKey).</li>
 50 	 *					<li>If the regex doesn't match any key, an empty array will be returned.</li>
 51 	 *				</ul>
 52 	 */
 53 	get: function(key, includeKey) {
 54 		if(Foundation.Storage.isRegExp(key)) {
 55 			var o = [];
 56 			
 57 			var properties = Ti.App.Properties.listProperties();
 58 			
 59 			for(var i in properties) {
 60 				if(key.test(properties[i]) && properties[i].match(/_type$/) == null) {
 61 					if(includeKey) {						
 62 						o.push({key: properties[i], value: Foundation.PersistentStorage._get(key)});
 63 					}
 64 					else {
 65 						o.push(Foundation.PersistentStorage._get(key));
 66 					}
 67 				}
 68 			}
 69 			
 70 			return o;
 71 		}
 72 		else {
 73 			if(Ti.App.Properties.hasProperty(key) && Ti.App.Properties.hasProperty(key + '_type')) {
 74 				if(includeKey) return {key: key, value: Foundation.PersistentStorage._get(key)};
 75 				else {
 76 					var value = Foundation.PersistentStorage._get(key);
 77 					return value;
 78 				}
 79 			}
 80 			else return null;
 81 		}
 82 	},
 83 	
 84 	/**
 85 	 * Retrieves the value for the specified key in the correct format.
 86 	 * @private
 87 	 * @param {string} key	The dictionary's key
 88 	 * @return {mixed}	The return value or null if no value is found.
 89 	 */
 90 	_get: function(key) {
 91 		var value = null;
 92 		var type = Ti.App.Properties.getString(key + '_type');
 93 		
 94 		switch(type) {
 95 			case 'Bool':
 96 				value = Ti.App.Properties.getBool(key);
 97 			break;
 98 			case 'String':
 99 				value = Ti.App.Properties.getString(key);
100 			break;
101 			case 'List':
102 				value = Ti.App.Properties.getList(key);
103 			break;
104 			case 'Int':
105 				value = Ti.App.Properties.getInt(key);
106 			break;
107 			case 'Double':
108 				// you might wonder why this instead of getDouble(): well, it's more accurate
109 				// (always returns the correct type and value 100%, whilst getDouble() approximates the
110 				// value which therefore might not be correct in all the architectures)
111 				value = parseFloat(Ti.App.Properties.getString(key));
112 			break;
113 			case 'object':
114 				value = Ti.App.Properties.getString(key);
115 				value = JSON.parse(value);
116 			break;
117 		}
118 		
119 		return value;
120 		
121 	},
122 	
123 	/**
124 	 * Sets a value.
125 	 * @param {string} key The key
126 	 * @param {mixed} value The value
127 	 */
128 	set: function(key, value) {
129 		var type = Foundation.PersistentStorage.typeOf(value);
130 		
131 		switch(type) {
132 			case 'Bool':
133 				Ti.App.Properties.setBool(key, value);
134 			break;
135 			case 'String':
136 				Ti.App.Properties.setString(key, value);
137 			break;
138 			case 'List':
139 				Ti.App.Properties.setList(key, value);
140 			break;
141 			case 'Int':
142 				Ti.App.Properties.setInt(key, value);
143 			break;
144 			case 'Double':
145 				Ti.App.Properties.setDouble(key, value);
146 			break;
147 			case 'object':
148 				value = JSON.stringify(value);
149 				Ti.App.Properties.setString(key, value);
150 			break;
151 		}
152 		
153 		Ti.App.Properties.setString(key + '_type', type);
154 		
155 		Foundation.Storage.checkWatchKey(key, value);
156 		
157 		return value;
158 	},
159 	
160 	/**
161 	 * Determines the correct type of the specified value.
162 	 * @private
163 	 * @param {mixed} value The value to check
164 	 * @return {string} A string containing the value type (Bool, String, object, List, Double, Int).
165 	 * 				    The name format allows other method to use reflection (Ti.App.Properties.getXXXX)
166 	 */
167 	typeOf: function(value) {
168 		var type = null;
169 		
170 		switch(typeof value) {
171 			case 'boolean':
172 				type = 'Bool';
173 			break;
174 			case 'string':
175 				type = 'String';
176 			break;
177 			case 'object':
178 				// value instanceof Array does not report the correct result in Titanium.
179 				if(value.constructor && value.constructor.toString().match(/^function Array\(\)/) != null) {
180 					type = 'List';
181 				}
182 				else {
183 					type = 'object';
184 				}
185 			break;
186 			case 'number':
187 				type = parseInt(value) != value ? 'Double' : 'Int';
188 			break;
189 		}
190 		
191 		return type;
192 	},
193 		
194 	/**
195 	 * Permanently removes a value from the storage.
196 	 * @param {mixed} key The key to remove. If string, the key will be removed, if present. You can
197 	 *					   provide a regex that will delete all the matching keys.
198 	 */
199 	remove: function(key) {
200 		if(Foundation.Storage.isRegExp(key)) {
201 			
202 			var properties = Ti.App.Properties.listProperties();
203 			
204 			for(var i in properties) {
205 				if(key.test(properties[i]) && properties[i].match(/_type$/) == null) {
206 					Foundation.Storage.checkWatchKey(key, null);
207 					Ti.App.Properties.removeProperty(properties[i]);
208 					Ti.App.Properties.removeProperty(properties[i] + '_type');
209 				}
210 			}
211 		}
212 		else if(Ti.App.Properties.hasProperty(key)) {
213 			Foundation.Storage.checkWatchKey(key, null);
214 			Ti.App.Properties.removeProperty(key);
215 		}	
216 	},
217 	
218 	/**
219 	 * Resets the storage.
220 	 */
221 	reset: function() {
222 		var properties = Ti.App.Properties.listProperties();
223 		for(var i in properties) {
224 			Ti.App.Properties.removeProperty(properties[i]);
225 		}
226 	},
227 
228 	/**
229 	 * Start watching one or more keys for any change. Use this method when you want your application to be
230 	 * notifies when a value is added, changed or removed (via Ti.App.addEventListener).
231 	 * @param {mixed} key	The key to watch. It can be a regex or a string.
232 	 */
233 	watch: function(key) {
234 		Foundation.PersistentStorage._watch[key.toString()] = (Foundation.Storage.isRegExp(key) ? 'regex' : 'string');
235 	},
236 	
237 	/**
238 	 * Stop watching one or more keys.
239 	 * @param {mixed} key	The key to watch. It can be a regex or a string.
240 	 */
241 	unwatch: function(key) {
242 		delete Foundation.PersistentStorage._watch[key.toString()];
243 	},
244 	
245 	/**
246 	 * Checks if any key listener is registered against the specified key.
247 	 *
248 	 * First of all, this method checks if the key dictionary contains a matching key (either a regex or a string).
249 	 * A regex is syntesized by using its string value (as we can't store regex references properly). If any matching
250 	 * key is found, this method will trigger an event.
251 	 * @private
252 	 * @param {mixed} key The value key (either a regex or a string)
253 	 * @param {mixed} value The value for the specified key. It will be passed along with the event data.
254 	 */
255 	checkWatchKey: function(key, value) {
256 		if(Foundation.PersistentStorage._watch[key.toString()]) {
257 			if(Foundation.PersistentStorage._watch[key.toString()] == 'regex') {
258 				regex = new RegExp(key.toString());
259 				if(regex.test(key)) {
260 					Ti.App.fireEvent(Foundation.PersistentStorage.Event.CHANGE, {type:'set', keySelector: key.toString(), key:key.toString(), value: value});
261 				}
262 			}			
263 			else if(Foundation.PersistentStorage._watch[key.toString()] == 'string') {
264 				Ti.App.fireEvent(Foundation.PersistentStorage.Event.CHANGE, {type:'set', keySelector: key.toString(), key:key.toString(), value: value});					
265 			}
266 		}
267 	}
268 };
