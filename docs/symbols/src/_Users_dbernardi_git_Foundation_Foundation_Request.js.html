1 /**
  2  * Performs XHR requests using Ti.Network.HTTPClient.
  3  * @class
  4  */
  5 Foundation.Request = /** @lends Foundation.Request# */ {
  6 	Event: {LOAD:'Foundation.Request.Event.LOAD', ERROR: 'Foundation.Request.Event.ERROR'},
  7 	CALLBACK_NAME: 'Foundation.Request.callback',
  8 		
  9 	// converts an object into a querystring
 10 	serialize: function(params, string, parent) {
 11 		string = string || '';
 12 		parent = parent || '';
 13 
 14 		// is array
 15 		for(var i in params) {
 16 			if(Foundation.isArray(params[i])) {
 17 				string = Foundation.Request.serialize(params[i], string, parent != '' ? '['+i+'][]': i + '[]') + '&';
 18 			}
 19 			else if(Foundation.isObject(params[i])) {
 20 				string = Foundation.Request.serialize(params[i], string, parent != '' ? '['+i+']': i) + '&';
 21 			}
 22 			else {
 23 				var p = "";
 24 				if(isNaN(i)) {
 25 					p = parent != '' ? parent + '['+i+']' : i;
 26 				}
 27 				else {
 28 					p = parent;
 29 				}
 30 				
 31 				p += '=';
 32 				string += p + params[i] + '&';				
 33 			}
 34 
 35 		}
 36 
 37 		string = string.replace(/\&$/,'');
 38 		return encodeURIComponent(string);
 39 	},
 40 	
 41 	// This method is executed as callback for a JSONP request
 42 	callback: function(response) { return JSON.parse(response); },
 43 	
 44 	/**
 45 	 * Prepares a JSON/JSONP request using the specified method.
 46 	 * This method internally proxies Titanium.HTTPClient, so further scrambling can be made, which means that:
 47 	 * <ul>
 48 	 * <li>This method performs a request and parses the result to get a valid JSON object, keeping the original
 49 	 * response.</li>
 50 	 * <li>This method can perform a JSONP request with a specified callback name, or use the default one.</li>
 51 	 * </ul>
 52 	 * This method will return a proxy to an instance of Titanium.HTTPClient, which means that you will still be able
 53 	 * to use the same (documented) entry points you should expect to find on that object.
 54 	 * 
 55 	 * @param	{String}	url		The request URL
 56 	 * @param	{String}	[type="GET"]		The type of request. So far, GET and POST are been tested successfully. Titanium
 57 	 *									should support PUT and DELETE, although these methods haven't been tested.
 58 	 * @param	{Object}	[options]		A dictionary of options specific to Foundation.Request
 59 	 * @param	{String}	[options.user]	Username for HTTP basic authentication
 60 	 * @param	{String}	[options.pass]	Password for HTTP basic authentication
 61 	 * @param	{Boolean}	[options.authentication=false] Enables or disabled basic authentication for the current request.
 62 	 *									If true, you must provide username and password.
 63 	 * @param	{Boolean|String} [options.jsonCallback] True to use Foundation's internal callback (Foundation.Request.callback). You can provide a function name
 64 	 *									if you need a different callback. The function must accept the response body as parameter.
 65 	 * @return {Object}					A proxy to Ti's HTTPClient. You should expect to use the same entry points as it.
 66 	 */
 67 	json: function(url, type, options) {
 68 		
 69 		options = options || {};
 70 		
 71 		var user = options.user || "";
 72 		var pass = options.pass || "";
 73 		
 74 		var jsonp = "";
 75 		
 76 		if(options.jsonpCallback) {
 77 			if(options.jsonpCallback === true) {
 78 				jsonp = 'callback=' + Foundation.Request.CALLBACK_NAME;
 79 			}
 80 			else if(typeof options.jsonpCallback == 'string') {
 81 				jsonp = 'callback=' + options.jsonpCallback;
 82 			}
 83 		}
 84 				
 85 		type = type || 'GET';
 86 		
 87 		var req = Ti.Network.createHTTPClient();
 88 		
 89 		req.open(type, url + (jsonp != '' ? '?' : '') + jsonp);
 90 
 91 		if(options.authentication) {
 92 			req.setRequestHeader('Authorization', 'Basic ' + Ti.Utils.base64encode(user+':'+pass));			
 93 		}
 94 
 95 		// Proxy for HTTPClient.
 96 		return new function FoundationTiHTTPClientProxy() {
 97 			this.timeout = 0;
 98 			// wraps the original setTimeout so that I can store the user's timeout
 99 			this.setTimeout = function(msec) {this.timeout = msec; req.setTimeout(msec);}
100 			
101 			// initialize onload and onerror to empty functions so it won't fail layer if the user doesn't specify any
102 			this.onload = function() {};
103 			this.onerror = function() {};
104 			
105 			// wraps the original send method, so I can pass an object to it and Titanium won't complain
106 			this.send = function(data) {
107 
108 				// if a GET request and we have a dictionary of parameters, convert them into a query string
109 				if(data && type.toUpperCase() == 'GET' && typeof data != 'string') {
110 					params = [];
111 					for(var i in data) {
112 						params.push(i + '=' + data[i]);
113 					}
114 					
115 					data = (jsonp != '' ? '&' : '') + params.join('&');
116 				}
117 				
118 				// if GET, I need to reconfigure the request because I need to append the query string,
119 				// so let's reopen the connection
120 				if(type.toUpperCase() == 'GET') {
121 					data = data || "";
122 					req.open(type, url + (jsonp != '' || data != '' ? '?' : '') + jsonp + data);					
123 					if(this.timeout > 0) {
124 						req.setTimeout(this.timeout);
125 					}
126 
127 					if(options.authentication) {
128 						req.setRequestHeader('Authorization', 'Basic ' + Ti.Utils.base64encode(user+':'+pass));			
129 					}
130 				}
131 
132 				// don't put any data into the request body if GET, otherwise Titanium will automatically
133 				// transform it into a POST request
134 				if(type.toUpperCase() == 'GET') {
135 					req.send();									
136 				}
137 				else {
138 					req.send(data || null);
139 				}
140 
141 			}
142 
143 			var self = this;
144 			// delegate events
145 			req.onerror = function() { self.onerror.apply(this, arguments); };
146 			req.onload = function() {
147 
148 				// responseJSON will be available in the function body and hopefully should
149 				// contain the response body in JSON format ready to be used
150 				try {
151 					this.responseJSON = JSON.parse(this.responseText);
152 				} catch(e) {
153 					Ti.API.error('[Foundation.Request.json] Exception while parsing JSON response: ' + e);
154 					this.responseJSON = null;
155 				}
156 
157 				self.onload.call(this, arguments);
158 			};
159 			
160 			return this;
161 		};
162 	}
163 };
