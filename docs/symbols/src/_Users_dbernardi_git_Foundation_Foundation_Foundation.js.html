 23 Foundation.platformDir = Foundation.Platform.isIOS() ? Foundation.Platform.IOS_DIR : Foundation.Platform.ANDROID_DIR;

 26 /**
 27  * Extends objects (literals). Useful where you need to add properties to an object or to override
 28  * properties or methods. This method is not recursive (i.e. only copies plain objects or references)
 29  * and does not overwrite its arguments.
 30  *
 31  * @param {Object} [object] The first object
 32  * @param {Object[]} [...] Optional objects
 33  * @return {Object} The augmented object.
 34  */
 35 Foundation.augment = function() {
 36 	var base = {};
 37 	for(var i = 0; i < arguments.length; i++) {
 38 	
 39 		src = arguments[i];
 40 
 41 		if(typeof src == 'object' && src !== null) {
 42 
 43 			for(var key in src) {
 44 				base[key] = src[key];
 45 			}
 46 	
 47 		}		
 48 	}
 49 	
 50 	return base;
 51 };

 53 /**
 54  * Provides interaction with tabs and windows.
 55  * @class
 56  */
 57 Foundation.UI = /** @lends Foundation.UI# */ {
 58 	
 59 	windowConfig: {},
 60 	windowTemplate: {},
 61 	
 62 	/**
 63 	 * Creates a new window as you would expect Titanium to do, but appends Foundation and the app namespace
 64 	 * to it. Also, this method will choose the right view to use for the window, provided that you have custom views.
 65 	 * If you want to use a custom view (sometimes referred as platform-specific views), create the path Resources/iphone/Views
 66 	 * and/or Resources/android/Views. Those paths will be checked first. If no custom view is found there, Foundation will use
 67 	 * the default path (Resources/Views).
 68 	 *
 69 	 * @param {string} name	The window name. It will be used for internal reference, as well to
 70 	 *						fetch the file that contains the view's code. To comply to Foundation's naming
 71 	 * 						conventions, any space will be trimmed. So, if your window title is “Apples and 
 72 	 *						Oranges”, this method will fetch the file ApplesAndOranges.js.
 73 	 * @param {object} params The configuration options, same as Titanium's parameters.
 74 	 * @param {boolean} [params.viewless] true to create a window without loading its relative file
 75 	 * @param {string} [params.file] the file name without path (must be under Resources/Views) that contains
 76      *								 the code for this window
 77 	 * @return {Ti.UI.Window} A Window instance
 78 	 */
 79 	createWindow: function(name, params) {
 80 		
 81 		params = params || {};
 82 		
 83 		var filename = '';
 84 		
 85 		if(!params.viewless) {
 86 			filename = (params.file || name.replace(/ /g, '') + '.js');
 87 
 88 			var f = Ti.Filesystem.getFile(Ti.Filesystem.resourcesDirectory + '/' + Foundation.platformDir + '/Views/' + filename);
 89 			if(f.exists()) {
 90 				filename = f.nativePath;
 91 			}
 92 			else {
 93 				filename = Foundation.prefix + 'Views/' + filename;
 94 			}
 95 		}
 96 		
 97 		var config = Foundation.UI.windowConfig[name] = {
 98 			title: name,
 99 			backgroundColor: '#fff',
100 			url: filename
101 		};
102 
103 		if(params) {
104 			config = Foundation.augment(config, Foundation.UI.windowTemplate, params);
105 		}
106 		
107 		if(config.viewless) {
108 			delete config.url;
109 			delete Foundation.UI.windowConfig[name].url;
110 		}
111 		
112 		Foundation.Windows[name] = Ti.UI.createWindow(config);	
113 
114 		// append instance strictly *after* creating the window,
115 		// otherwise will crash
116 		Foundation.Windows[name].app = app;
117 		Foundation.Windows[name].id = name;
118 		Foundation.Windows[name].f = Foundation.Windows[name].foundation = Foundation;
119 
120 		return Foundation.Windows[name];
121 	},
122 	
123 	/**
124 	 * Returns the specified window, if exists.
125 	 * @param {string} name	The window name.
126 	 * @param {boolean} [withConfig=false] to return a dictionary with the window's current parameters.
127 	 * @return {mixed} a Ti.UI.Window object or, if withConfig set to true, a dictionary with the
128 	 * 					properties win (the window instance) and config (the window configuration).
129 	 */
130 	getWindow: function(name, withConfig) {
131 		if(Foundation.Windows[name]) {
132 			return withConfig ? {win: Foundation.Windows[name], config: Foundation.UI.windowConfig[name]} : Foundation.Windows[name];
133 		}
134 		else {
135 			return null;
136 		}
137 	},
138 	
139 	/**
140 	 * Returns the current window template, if set.
141 	 * @return {object} The window template. An empty object literal will be returned if no template is set.
142 	 */
143 	getWindowTemplate: function() { return Foundation.UI.windowTemplate; },
144 
145 	/**
146 	 * Applies and stores a window template.
147 	 * @param {object} template The template to apply.
148 	 */
149 	setWindowTemplate: function(template) {
150 		
151 		for(var name in Foundation.Windows) {
152 			for(var i in template) {
153 				Foundation.Windows[name][i] = template[i];
154 			}
155 		}				
156 		
157 		Foundation.UI.windowTemplate = template;
158 	},
159 
160 	/**
161 	 * Opens a new window and creates it if needed. Internally calls Foundation.UI.createWindow, so it works
162 	 * in a similar way. You should always prefer this method over createwindow, unless you don't need to open
163 	 * the window right away.
164 	 *
165 	 * @param {string} name	The name to assign to the window. It will be used for internal reference and to
166 	 *						fetch the file that contains the views's code.
167 	 * @param {mixed} context A Ti.UI.TabbedBar or Ti.UI.iPhone.NavigationGroup object in which the window
168 	 *						  should be open
169 	 * @param {object} [windowOptions] A dictionary of properties as Ti.UI.createWindow would expect
170 	 * @param {object} [options] A dictionary of properties as Ti.UI.TabbedBar or Ti.UI.iPhone.NavigationGroup
171 	 * 							 would expect.
172 	 */
173 	openWindow: function(name, context, windowOptions, options) {
174 		
175 		windowOptions = windowOptions || null;
176 		options = options || {animated:true};
177 
178 		Foundation.Windows[name] = Foundation.UI.createWindow(name, windowOptions || null);
179 
180 		// context should be Ti.UI.currentTab or a Navigation Group instance
181 		context.open(Foundation.Windows[name], options);
182 	},
183 	
184 	/**
185 	 * Creates a new tab and appends it to the tab group. For the moment only one tab group is allowed in a
186 	 * Foundation application, so it will be automatically created if needed.
187 	 *
188 	 * @param {string} name	The name to assign to the window. It will be used for internal reference and to
189 	 *						fetch the file that contains the views's code.
190 	 * @param {object} [tabOptions] A dictionary of properties as Ti.UI.createTab would expect
191 	 * @param {object} [winOptions] A dictionary of properties as Foundation.UI.createWindow would expect
192 	 * @return {Ti.UI.Tab} The tab
193 	 */
194 	createTab: function(name, tabOptions, winOptions) {
195 		
196 		tabOptions = tabOptions || {};
197 		winOptions = winOptions || null;
198 		
199 		// create a tabgroup if none has been created before
200 		if(Foundation.currentTabGroup === null) {
201 			Foundation.UI.createTabGroup();
202 		}
203 
204 		// creates a window if doesn't exists. The window will have the same name
205 		// as the tab.
206 		if (typeof name == 'string' && !Foundation.Windows[name]) {
207 			var win = Foundation.UI.createWindow(name, winOptions);
208 		}
209 		
210 		var options = {
211 			title: name,
212 			backgroundColor: '#444'
213 		};
214 
215 		var icon = Ti.Filesystem.getFile(Ti.Filesystem.resourcesDirectory + '/Images/' + name.replace(/ /g, '') + '.png');
216 
217 		if(icon.exists()) {
218 			options.icon = icon.nativePath;
219 		}
220 		else {
221 			options.icon = Foundation.prefix + 'Images/FoundationGenericIcon.png';
222 		}
223 
224 		tabOptions = Foundation.augment(options, tabOptions);
225 		tabOptions.window = tabOptions.window || Foundation.Windows[name];
226 		
227 		Foundation.Tabs[name] = Ti.UI.createTab(tabOptions);
228 		Foundation.Tabs[name].id = name;
229 		Foundation.Tabs[name].mainWindowId = name;
230 		Foundation.UI.tabGroup().addTab(Foundation.Tabs[name]);
231 		
232 		return Foundation.Tabs[name];
233 	},
234 
235 	/**
236 	 * Return the specifies tab group.
237 	 * @param {string} [tabGroupId] The tab group identifier. Leave empty to return the current active tabgroup.
238 	 * @return {Ti.UI.TabGroup} The current TabGroup or null of the specified tabgroup is not present.
239 	 */
240 	tabGroup: function(tabGroupId) {
241 		tabGroupId = tabGroupId || Foundation.currentTabGroup;
242 		return Foundation.TabGroups[tabGroupId]; 
243 	},
244 	
245 	/**
246 	 * Closes the current active tabgroup and opens the specified tabgroup.
247 	 * @param {string} [tabGroupId] The tab group identifier. Leave empty to close and reopen the current active tabgroup;
248 	 */
249 	openTabGroup: function(tabGroupId) {
250 		
251 		tabGroupId = tabGroupId || Foundation.currentTabGroup;
252 		
253 		Foundation.TabGroups[Foundation.currentTabGroup].close();
254 		Foundation.TabGroups[tabGroupId].open();
255 		
256 		Foundation.currentTabGroup = tabGroupId;
257 	},
258 	
259 	/**
260 	 * Creates a tab group.
261 	 * @param {string} [tabGroupId] The tab group identifier.
262 	 * @param {object} [params] The params to give to the tab group, as Ti.UI.createTabGroup would expect.
263 	 * @returns {Ti.UI.tabGroup} A Ti.UI.tabGroup instance
264 	 */
265 	createTabGroup: function(tabGroupId, params) {
266 
267 		params = params || {};
268 		
269 		if(tabGroupId instanceof Object) {
270 			params = tabGroupId;
271 			tabGroupId = Foundation.tabGroupCounter;
272 			Foundation.tabGroupCounter++;
273 		}
274 		else if(!tabGroupId) {
275 			tabGroupId = Foundation.tabGroupCounter;
276 			Foundation.tabGroupCounter++;
277 		}
278 		
279 		Foundation.TabGroups[tabGroupId] = Ti.UI.createTabGroup(params);
280 		
281 		if(Foundation.currentTabGroup === null) {
282 			Foundation.currentTabGroup = tabGroupId;
283 		}
284 		
285 		return Foundation.TabGroups[tabGroupId];
286 	}
287 };
