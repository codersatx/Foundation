 18 /**
 19  * Extends objects (literals). Useful where you need to add properties to an object or to override
 20  * properties or methods. This method is not recursive (i.e. only copies plain objects or references)
 21  * and does not overwrite its arguments.
 22  *
 23  * @param {Object} [object] The first object
 24  * @param {Object[]} [...] Optional objects
 25  * @return {Object} The augmented object.
 26  */
 27 Foundation.augment = function() {
 28 	var base = {};
 29 	for(var i = 0; i < arguments.length; i++) {
 30 	
 31 		src = arguments[i];
 32 
 33 		if(typeof src == 'object' && src !== null) {
 34 
 35 			for(var key in src) {
 36 				base[key] = src[key];
 37 			}
 38 	
 39 		}		
 40 	}
 41 	
 42 	return base;
 43 };

 45 /**
 46  * Provides interaction with tabs and windows.
 47  * @class
 48  */
 49 Foundation.UI = /** @lends Foundation.UI# */ {
 50 	
 51 	windowConfig: {},
 52 	
 53 	/**
 54 	 * Creates a new window as you would expect Titanium to do, but appends Foundation and the app namespace
 55 	 * to it.
 56 	 * @param {string} name	The window name. It will be used for internal reference, as well to
 57 	 *						fetch the file that contains the view's code. To comply to Foundation's naming
 58 	 * 						conventions, any space will be trimmed. So, if your window title is “Apples and 
 59 	 *						Oranges”, this method will fetch the file ApplesAndOranges.js.
 60 	 * @param {string} params The configuration options, same as Titanium's parameters.
 61 	 * @param {boolean} [params.viewless] true to create a window without loading its relative file
 62 	 * @param {string} [params.file] the file name without path (must be under Resources/Views) that contains
 63      *								 the code for this window
 64 	 * @return {Ti.UI.Window} A Window instance
 65 	 */
 66 	createWindow: function(name, params) {
 67 		
 68 		params = params || {};
 69 
 70 		var config = Foundation.UI.windowConfig[name] = {
 71 			title: name,
 72 			backgroundColor: '#fff',
 73 			url: 'Views/' + (params.file || name.replace(/ /g, '') + '.js')
 74 		};
 75 
 76 		if(params) {
 77 			config = Foundation.augment(config, params);
 78 		}
 79 		
 80 		if(config.viewless) {
 81 			delete config.url;
 82 			delete Foundation.UI.windowConfig[name].url;
 83 		}
 84 		
 85 		Foundation.Windows[name] = Ti.UI.createWindow(config);	
 86 
 87 		// append instance strictly *after* creating the window,
 88 		// otherwise will crash
 89 		Foundation.Windows[name].app = app;
 90 		Foundation.Windows[name].id = name;
 91 		Foundation.Windows[name].f = Foundation.Windows[name].foundation = Foundation;
 92 
 93 		return Foundation.Windows[name];
 94 	},
 95 	
 96 	/**
 97 	 * Returns the specified window, if exists.
 98 	 * @param {string} name	The window name.
 99 	 * @param {boolean} [withConfig=false] to return a dictionary with the window's current parameters.
100 	 * @return {mixed} a Ti.UI.Window object or, if withConfig set to true, a dictionary with the
101 	 * 					properties win (the window instance) and config (the window configuration).
102 	 */
103 	getWindow: function(name, withConfig) {
104 		if(Foundation.Windows[name]) {
105 			return withConfig ? {win: Foundation.Windows[name], config: Foundation.UI.windowConfig[name]} : Foundation.Windows[name];
106 		}
107 		else {
108 			return null;
109 		}
110 	},
111 
112 	/**
113 	 * Opens a new window and creates it if needed. Internally calls Foundation.UI.createWindow, so it works
114 	 * in a similar way. You should always prefer this method over createwindow, unless you don't need to open
115 	 * the window right away.
116 	 *
117 	 * @param {string} name	The name to assign to the window. It will be used for internal reference and to
118 	 *						fetch the file that contains the views's code.
119 	 * @param {mixed} context A Ti.UI.TabbedBar or Ti.UI.iPhone.NavigationGroup object in which the window
120 	 *						  should be open
121 	 * @param {object} [windowOptions] A dictionary of properties as Ti.UI.createWindow would expect
122 	 * @param {object} [options] A dictionary of properties as Ti.UI.TabbedBar or Ti.UI.iPhone.NavigationGroup
123 	 * 							 would expect.
124 	 */
125 	openWindow: function(name, context, windowOptions, options) {
126 		
127 		windowOptions = windowOptions || null;
128 		options = options || {animated:true};
129 
130 		Foundation.Windows[name] = Foundation.UI.createWindow(name, windowOptions || null);
131 
132 		// context should be Ti.UI.currentTab or a Navigation Group instance
133 		context.open(Foundation.Windows[name], options);
134 	},
135 	
136 	/**
137 	 * Creates a new tab and appends it to the tab group. For the moment only one tab group is allowed in a
138 	 * Foundation application, so it will be automatically created if needed.
139 	 *
140 	 * @param {string} name	The name to assign to the window. It will be used for internal reference and to
141 	 *						fetch the file that contains the views's code.
142 	 * @param {object} [tabOptions] A dictionary of properties as Ti.UI.createTab would expect
143 	 * @param {object} [winOptions] A dictionary of properties as Foundation.UI.createWindow would expect
144 	 * @return {Ti.UI.Tab} The tab
145 	 */
146 	createTab: function(name, tabOptions, winOptions) {
147 		
148 		tabOptions = tabOptions || {};
149 		winOptions = winOptions || null;
150 		
151 		// create a tabgroup if none has been created before
152 		if(Foundation.TabGroup == null) {
153 			Foundation.TabGroup = Ti.UI.createTabGroup();			
154 		}
155 
156 		// creates a window if doesn't exists. The window will have the same name
157 		// as the tab.
158 		if (typeof name == 'string' && !Foundation.Windows[name]) {
159 			var win = Foundation.UI.createWindow(name, winOptions);
160 		}
161 		
162 		var options = {
163 			title: name,
164 			backgroundColor: '#444'
165 		};
166 
167 		var icon = Ti.Filesystem.getFile(Ti.Filesystem.resourcesDirectory + '/Images/' + name.replace(/ /g, '') + '.png');
168 
169 		if(icon.exists()) {
170 			options.icon = icon.nativePath;
171 		}
172 		else {
173 			options.icon = 'Images/FoundationGenericTab.png';
174 		}
175 
176 		tabOptions = Foundation.augment(options, tabOptions);
177 		tabOptions.window = tabOptions.window || Foundation.Windows[name];
178 		
179 		Foundation.Tabs[name] = Ti.UI.createTab(tabOptions);
180 		Foundation.Tabs[name].id = name;
181 		Foundation.Tabs[name].mainWindowId = name;
182 		Foundation.TabGroup.addTab(Foundation.Tabs[name]);
183 		
184 		return Foundation.Tabs[name];
185 	},
186 
187 	/**
188 	 * Return the current tab group.
189 	 * @return {Ti.UI.TabGroup} The current TabGroup
190 	 */
191 	tabGroup: function() { return Foundation.TabGroup; }
192 };
