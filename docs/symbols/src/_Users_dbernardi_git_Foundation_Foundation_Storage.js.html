1 /**
  2  * Dictionary-based storage for runtime data.
  3  * @class
  4  */
  5 Foundation.Storage = /** @lends Foundation.Storage# */{
  6 
  7 	/**
  8 	 * @name Foundation.Storage#change
  9 	 * @event
 10 	 * @param {string} type The type of event. It can be 'set' (value added for the first time or changed) or 'remove'
 11 	 *						(value no longer present into Storage).
 12 	 * @param {string} keySelector The registered key listered that triggered the event (e.g. /^SomeExampleKey[0-9]+/).
 13 	 *						       Note that both RegExp and string key listeners will be passed as strings here.
 14 	 * @param {string} key The key of the element for which the event was triggered (e.g 'SomeExampleKey1234')
 15 	 */
 16 	
 17 	_data: {},
 18 	_watch: {},
 19 
 20 	/**
 21 	 * Gets one or more values safely.
 22 	 * @param {mixed} key The key to remove. If string, the key will be removed, if present. You can
 23 	 *					   provide a regex that will delete all the matching keys.
 24 	 * @param {boolean} [includeKey=false] True to return an object with two properties: key (the key for this value)
 25 	 *                  and value (the value).
 26 	 * @return {mixed} If you provided a string key:
 27 	 * 				<ul>
 28 	 *					<li>If the key is present, the value will be returned.</li>
 29 	 *					<li>If the key is present and includeKey is true, an object will be returned (see the description
 30 	 *                  for includeKey).</li>
 31 	 *					<li>If the key is not present, null will be returned.</li>
 32 	 *				</ul>
 33 	 *				If you provided a regex:
 34 	 * 				<ul>
 35 	 *					<li>If the regex matches one or more keys, an array of values will be returned.</li>
 36 	 *					<li>If the regex matches one or more keys and includeKey is true, an array of objects
 37 	 *                  will be returned (see the description for includeKey).</li>
 38 	 *					<li>If the regex doesn't match any key, an empty array will be returned.</li>
 39 	 *				</ul>
 40 	 */
 41 	get: function(key, includeKey) {
 42 		if(key instanceof RegExp) {
 43 			var o = [];
 44 			for(var i in Foundation.Storage._data) {
 45 				if(key.test(i)) {
 46 					if(includeKey) {
 47 						o.push({key: i, value: Foundation.Storage._data[i]});
 48 					}
 49 					else {
 50 						o.push(Foundation.Storage._data[i]);						
 51 					}
 52 				}
 53 			}
 54 			
 55 			return o;
 56 		}
 57 		else {
 58 			if(Foundation.Storage._data[key]) {
 59 				if(includeKey) return {key: key, value: Foundation.Storage._data[key]};
 60 				else return Foundation.Storage._data[key];
 61 				
 62 			}
 63 			else return null;
 64 		}
 65 	},
 66 	
 67 	/**
 68 	 * Sets a value.
 69 	 * @param {string} key The key
 70 	 * @param {mixed} value The value
 71 	 */
 72 	set: function(key, value) {
 73 		Foundation.Storage._data[key] = value;
 74 		
 75 		Foundation.Storage.checkWatchKey(key, value, 'set');
 76 		
 77 		return Foundation.Storage._data[key];
 78 	},
 79 		
 80 	/**
 81 	 *	@param {mixed} key The key to remove. If string, the key will be removed, if present. You can
 82 	 *					   provide a regex that will delete all the matching keys.
 83 	 */
 84 	remove: function(key) {
 85 		if(key instanceof RegExp) {
 86 			for(var i in Foundation.Storage._data) {
 87 				if(key.test(i)) {
 88 					delete Foundation.Storage._data[i];
 89 					
 90 					Foundation.Storage.checkWatchKey(i, null, 'remove');
 91 				}
 92 			}
 93 		}
 94 		else if(Foundation.Storage._data[key]) {
 95 			value = Foundation.Storage._data[key];
 96 			delete Foundation.Storage._data[key];
 97 			
 98 			Foundation.Storage.checkWatchKey(key, null, 'remove');
 99 			
100 		}	
101 	},
102 	
103 	/**
104 	 * Wipes the storage.
105 	 * @param {boolean} [alsoResetKeyListeners=false] Removes all the key listeners.
106 	 */
107 	reset: function(alsoResetKeyListeners) {
108 		Foundation.Storage._data = {};
109 
110 		if(alsoResetKeyListeners) {
111 			Foundation.Storage._watch = {};
112 		}
113 	},
114 	
115 	/**
116 	 * Attaches a key listener.
117      * Every time that a value corresponding to the specified key is changed (that is, set or removed), a Foundation.Storage.change
118 	 * event is fired (through Ti.App.fireEvent), so you can attach events and perform actions on change.
119 	 * @param {string|RegExp} key The key to watch
120 	 *
121 	 */
122 	watch: function(key) {
123 		Foundation.Storage._watch[key.toString()] = key;
124 	},
125 
126 	/**
127 	 * Removes a key listener.
128 	 * @param {string|RegExp} key The key to remove
129 	 */
130 	unwatch: function(key) {
131 		delete Foundation.Storage._watch[key.toString()];
132 	},
133 	
134 	checkWatchKey: function(srcKey, value, type) {
135 		for(var key in Foundation.Storage._watch) {
136 			
137 			if(Foundation.Storage._watch[key] instanceof RegExp && Foundation.Storage._watch[key].test(srcKey)) {
138 				Ti.App.fireEvent('Foundation.Storage.change', {type:type, keySelector: key.toString(), key:srcKey, value: value});					
139 			}			
140 			else if(Foundation.Storage._watch[key] == srcKey) {
141 				Ti.App.fireEvent('Foundation.Storage.change', {type:type, keySelector: key.toString(), key:srcKey, value: value});					
142 			}
143 		}
144 	}
145 };
