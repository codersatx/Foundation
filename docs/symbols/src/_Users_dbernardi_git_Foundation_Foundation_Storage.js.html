1 /**
  2  * Dictionary-based storage for runtime data.
  3  */
  4 Foundation.Storage = /** @lends Foundation.Storage# */{
  5 	_data: {},
  6 	_watch: {},
  7 	Event: /** @lends Foundation.Storage.Event# */ {
  8 		/**
  9 		 * Event name to use when watching value changes.
 10 		 * @constant
 11 		 */
 12 		CHANGE: 'Foundation.Storage.change'
 13 	},
 14 	
 15 	/**
 16 	 * Gets one or more values safely.
 17 	 * @param {mixed} key The key to remove. If string, the key will be removed, if present. You can
 18 	 *					   provide a regex that will delete all the matching keys.
 19 	 * @param {boolean} [includeKey=false] True to return an object with two properties: key (the key for this value)
 20 	 *                  and value (the value).
 21 	 * @return {mixed} If you provided a string key:
 22 	 * 				<ul>
 23 	 *					<li>If the key is present, the value will be returned.</li>
 24 	 *					<li>If the key is present and includeKey is true, an object will be returned (see the description
 25 	 *                  for includeKey).</li>
 26 	 *					<li>If the key is not present, null will be returned.</li>
 27 	 *				</ul>
 28 	 *				If you provided a regex:
 29 	 * 				<ul>
 30 	 *					<li>If the regex matches one or more keys, an array of values will be returned.</li>
 31 	 *					<li>If the regex matches one or more keys and includeKey is true, an array of objects
 32 	 *                  will be returned (see the description for includeKey).</li>
 33 	 *					<li>If the regex doesn't match any key, an empty array will be returned.</li>
 34 	 *				</ul>
 35 	 */
 36 	get: function(key, includeKey) {
 37 		if(Foundation.Storage.isRegExp(key)) {
 38 			var o = [];
 39 			for(var i in Foundation.Storage._data) {
 40 				if(key.test(i)) {
 41 					if(includeKey) {
 42 						o.push({key: i, value: Foundation.Storage._data[i]});
 43 					}
 44 					else {
 45 						o.push(Foundation.Storage._data[i]);						
 46 					}
 47 				}
 48 			}
 49 			
 50 			return o;
 51 		}
 52 		else {
 53 			if(Foundation.Storage._data[key]) {
 54 				if(includeKey) return {key: key, value: Foundation.Storage._data[key]};
 55 				else return Foundation.Storage._data[key];
 56 				
 57 			}
 58 			else return null;
 59 		}
 60 	},
 61 	
 62 	isRegExp: function(key) { return key.constructor.toString().match(/^function RegExp\(\)/); },
 63 	
 64 	/**
 65 	 * Sets a value.
 66 	 * @param {string} key The key
 67 	 * @param {mixed} value The value
 68 	 */
 69 	set: function(key, value) {
 70 		Foundation.Storage._data[key] = value;
 71 		
 72 		Foundation.Storage.checkWatchKey(key, value);
 73 		
 74 		return Foundation.Storage._data[key];
 75 	},
 76 		
 77 	/**
 78 	 *	@param {mixed} key The key to remove. If string, the key will be removed, if present. You can
 79 	 *					   provide a regex that will delete all the matching keys.
 80 	 */
 81 	remove: function(key) {
 82 		if(Foundation.Storage.isRegExp(key)) {
 83 			for(var i in Foundation.Storage._data) {
 84 				if(key.test(i)) {
 85 					delete Foundation.Storage._data[i];
 86 					
 87 					Foundation.Storage.checkWatchKey(key, null);					
 88 				}
 89 			}
 90 		}
 91 		else if(Foundation.Storage._data[key]) {
 92 			value = Foundation.Storage._data[key];
 93 			delete Foundation.Storage._data[key];
 94 			
 95 			Foundation.Storage.checkWatchKey();
 96 			
 97 		}	
 98 	},
 99 	
100 	/**
101 	 * Resets the storage.
102 	 */
103 	reset: function() {
104 		Foundation.Storage._data = {};
105 	},
106 	
107 	/**
108 	 * Start watching one or more keys for any change. Use this method when you want your application to be
109 	 * notifies when a value is added, changed or removed (via Ti.App.addEventListener).
110 	 * @param {mixed} key	The key to watch. It can be a regex or a string.
111 	 */
112 	watch: function(key) {
113 		Foundation.Storage._watch[key.toString()] = (Foundation.Storage.isRegExp(key) ? 'regex' : 'string');
114 	},
115 	
116 	/**
117 	 * Stop watching one or more keys.
118 	 * @param {mixed} key	The key to watch. It can be a regex or a string.
119 	 */
120 	unwatch: function(key) {
121 		delete Foundation.Storage._watch[key.toString()];
122 	},
123 	
124 	checkWatchKey: function(key, value) {
125 		if(Foundation.Storage._watch[key.toString()]) {
126 			if(Foundation.Storage._watch[key.toString()] == 'regex') {
127 				regex = new RegExp(key.toString());
128 				if(regex.test(key)) {
129 					Ti.App.fireEvent(Foundation.Storage.Event.CHANGE, {type:'set', keySelector: key.toString(), key:key.toString(), value: value});					
130 				}
131 			}			
132 			else if(Foundation.Storage._watch[key.toString()] == 'string') {
133 				Ti.App.fireEvent(Foundation.Storage.Event.CHANGE, {type:'set', keySelector: key.toString(), key:key.toString(), value: value});					
134 			}
135 		}
136 	}
137 };
